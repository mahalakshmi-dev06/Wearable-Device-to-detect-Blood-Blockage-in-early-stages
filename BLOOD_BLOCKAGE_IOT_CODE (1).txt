/* Buffers */
#define FILTER_SIZE 8
long irBuffer[FILTER_SIZE];
long redBuffer[FILTER_SIZE];
int bufIndex=0;

/* Status */
String blockageStatus="PLACE FINGER";

float glucose=0;
String glucoseStatus="WAIT";

float hemoglobin=0;
String hemoStatus="WAIT";

int bpm=0;

/* Finger */
bool fingerPresent=false;

/* Timing */
unsigned long fingerStart=0;
bool bpmReady=false;

/* Stability */
int stableCounter=0;
bool stableSignal=false;

/* History */
#define HISTORY 10
int bpmHist[HISTORY];
float gluHist[HISTORY];
float hemoHist[HISTORY];
String blkHist[HISTORY];

int idx=0;

/* Calibration factors */

float glucoseGain=1.05;
float hemoGain=0.95;

/* ================================================== */

float computeAverage(long *arr)
{
long sum=0;

for(int i=0;i<FILTER_SIZE;i++)
sum+=arr[i];

return (float)sum/FILTER_SIZE;
}

/* ================================================== */

float smoothSignal(float prev,float current)
{
float alpha=0.4;
return (alpha*current)+(1-alpha)*prev;
}

/* ================================================== */

float calculateSignalQuality(float ir,float red)
{

float diff = abs(ir-red);

float quality = map(diff,0,40000,100,10);

quality = constrain(quality,10,100);

return quality;

}

/* ================= SETUP ================= */

void setup(){

Serial.begin(115200);

Wire.begin(21,22);

display.begin(SSD1306_SWITCHCAPVCC,0x3C);

display.clearDisplay();
display.setTextColor(WHITE);

WiFi.softAP(ssid,password);

server.on("/",handleRoot);
server.on("/status",handleStatus);
server.begin();

particleSensor.begin(Wire);
particleSensor.setup();

particleSensor.setPulseAmplitudeRed(0x2A);
particleSensor.setPulseAmplitudeIR(0x2A);

}

/* ================= LOOP ================= */

void loop(){

server.handleClient();

irValue = particleSensor.getIR();
redValue = particleSensor.getRed();

/* Store to buffer */

irBuffer[bufIndex]=irValue;
redBuffer[bufIndex]=redValue;

bufIndex++;
if(bufIndex>=FILTER_SIZE) bufIndex=0;

/* Average */

float irAvg = computeAverage(irBuffer);
float redAvg = computeAverage(redBuffer);

/* Smooth */

irFiltered = smoothSignal(irFiltered,irAvg);
redFiltered = smoothSignal(redFiltered,redAvg);

/* Finger detection */

fingerPresent = (irFiltered > 50000);

if(fingerPresent){

stableCounter++;

if(stableCounter>5) stableSignal=true;

}
else{

stableCounter=0;
stableSignal=false;

}

/* BPM estimation */

if(stableSignal){

bpmReady=true;

bpm = map(irFiltered,50000,120000,60,110);

bpm = constrain(bpm,60,120);

}
else{

bpmReady=false;
bpm=0;

}

/* Ratio calculation */

float ratio = redFiltered/irFiltered;

/* Glucose */

glucose = (80 + ratio*110) * glucoseGain;

/* Hemoglobin */

hemoglobin = (10 + ratio*6) * hemoGain;

/* Blockage */

int perfusion = map(irFiltered,50000,120000,0,100);

if(perfusion>70) blockageStatus="NORMAL";
else if(perfusion>50) blockageStatus="SLIGHT";
else if(perfusion>30) blockageStatus="MEDIUM";
else blockageStatus="HIGH";

/* History */

blkHist[idx]=blockageStatus;
gluHist[idx]=glucose;
hemoHist[idx]=hemoglobin;
bpmHist[idx]=bpm;

idx=(idx+1)%HISTORY;

/* OLED */

display.clearDisplay();

display.setCursor(0,0);
display.print("BPM: ");
display.print(bpm);

display.setCursor(0,16);
display.print("Block: ");
display.print(blockageStatus);

display.setCursor(0,32);
display.print("Glu: ");
display.print(glucose,0);

display.setCursor(0,48);
display.print("Hb: ");
display.print(hemoglobin,1);

display.display();

delay(1000);

}
